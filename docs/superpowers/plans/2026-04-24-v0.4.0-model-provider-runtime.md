# v0.4.0 Model Provider Runtime Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `v0.4.0` model provider runtime so Cliq can run against OpenRouter, Anthropic, OpenAI, OpenAI-compatible endpoints, and local Ollama, with a streaming-capable runtime event foundation.

**Architecture:** Keep the JSON-over-text action protocol unchanged. Add a model config resolver and provider registry below the CLI, upgrade sessions to structured provider/model identity, and make `ModelClient.complete()` return a structured completion while optionally emitting model stream events. The runner remains the action loop owner and translates provider stream events into runtime events that do not expose raw JSON action tokens to the default CLI renderer.

**Tech Stack:** TypeScript, Node.js built-in `fetch`, Node test runner, existing Cliq runtime/session/workspace modules.

---

## Scope Check

This plan implements the merged spec at `docs/superpowers/specs/2026-04-24-v0.4.0-model-provider-runtime-design.md`.

Included:

- provider registry for `openrouter`, `anthropic`, `openai`, `openai-compatible`, and `ollama`
- model config resolution from CLI, workspace config, environment, and defaults
- structured session model identity and schema v4 migration
- structured `ModelCompletion`
- optional model stream events
- runtime events for model/tool/final/error lifecycle
- CLI wiring and docs updates

Not included:

- cross-provider handoff
- remote dynamic model discovery
- token and cost accounting
- provider-native tool calling
- image, audio, or video generation
- token-by-token final answer rendering

export type ModelProvider = {
  name: ProviderName;
  supportsStreaming: boolean;
  defaultBaseUrl: string | null;
  createClient(config: ResolvedModelConfig): ModelClient;
  getDefaultModel(): string | null;
  getKnownModels(): ModelDescriptor[];

      displayName: 'Claude Sonnet 4',
      capabilities: { ...TEXT_CHAT_CAPABILITIES }
    }
  ],
  openai: [
    {
      provider: 'openai',
      model: 'gpt-5.2',
      displayName: 'GPT-5.2',
      capabilities: { ...TEXT_CHAT_CAPABILITIES }
    }
  ],
  'openai-compatible': [],
  ollama: []
};

const providerDefaults: Record<ProviderName, { baseUrl: string | null; model: string | null; supportsStreaming: boolean }> = {
  openrouter: { baseUrl: DEFAULT_MODEL_BASE_URL, model: MODEL, supportsStreaming: true },
  anthropic: { baseUrl: 'https://api.anthropic.com', model: 'claude-sonnet-4-20250514', supportsStreaming: true },
  openai: { baseUrl: 'https://api.openai.com/v1', model: 'gpt-5.2', supportsStreaming: true },
  'openai-compatible': { baseUrl: null, model: null, supportsStreaming: true },
  ollama: { baseUrl: 'http://localhost:11434', model: null, supportsStreaming: true }
};

const factories = new Map<ProviderName, (config: ResolvedModelConfig) => ModelClient>();

export function isProviderName(value: string): value is ProviderName {
  return (PROVIDER_NAMES as readonly string[]).includes(value);
}

export function registerModelClientFactory(provider: ProviderName, factory: (config: ResolvedModelConfig) => ModelClient) {
  factories.set(provider, factory);
}

export function getModelProvider(name: ProviderName): ModelProvider {
  const defaults = providerDefaults[name];
  return {
    name,
    supportsStreaming: defaults.supportsStreaming,
    defaultBaseUrl: defaults.baseUrl,
    createClient(config) {
      const factory = factories.get(name);
      if (!factory) {
        throw new Error(`Model provider ${name} is not registered`);
      }
      return factory(config);
    },
    getDefaultModel() {
      return defaults.model;
    },
    getKnownModels() {
      return descriptors[name];
    }
  };
}

export function createGenericTextModelDescriptor(provider: ProviderName, model: string): ModelDescriptor {
  return {
    provider,
    model,
    displayName: model,
    capabilities: { ...TEXT_CHAT_CAPABILITIES }
  };
}

```ts
import type { ProviderName, ResolvedModelConfig, StreamingMode } from './types.js';
import { DEFAULT_MODEL_CONFIG, getModelProvider, isProviderName } from './registry.js';

export type PartialModelConfig = {
  provider?: string;
  model?: string;
  baseUrl?: string;
  streaming?: string;
};

export type ModelConfigInput = {
  workspace: {
    model?: PartialModelConfig;
  };
  cli: PartialModelConfig;
};

function isStreamingMode(value: string): value is StreamingMode {
  return value === 'auto' || value === 'on' || value === 'off';
}

function firstDefined(...values: Array<string | undefined>) {
  return values.find((value) => value !== undefined && value !== '');
}

function providerApiKey(provider: ProviderName) {
  if (provider === 'openrouter') return process.env.OPENROUTER_API_KEY;
  if (provider === 'anthropic') return process.env.ANTHROPIC_API_KEY;
  if (provider === 'openai') return process.env.OPENAI_API_KEY;
  if (provider === 'openai-compatible') return process.env.CLIQ_MODEL_API_KEY;
  return undefined;
}

function requireKey(provider: ProviderName, apiKey: string | undefined) {
  if (provider === 'ollama') return;
  if (provider === 'openai-compatible') return;
  if (!apiKey) {
    throw new Error(`${provider} API key is required`);
  }
}

export function resolveModelConfig({ workspace, cli }: ModelConfigInput): ResolvedModelConfig {
  const rawProvider = firstDefined(cli.provider, workspace.model?.provider, process.env.CLIQ_MODEL_PROVIDER, DEFAULT_MODEL_CONFIG.provider);
  if (!rawProvider || !isProviderName(rawProvider)) {
    throw new Error(`Unknown model provider: ${rawProvider ?? ''}`);
  }

  const provider = rawProvider;
  const providerDef = getModelProvider(provider);
  const rawStreaming = firstDefined(cli.streaming, workspace.model?.streaming, process.env.CLIQ_MODEL_STREAMING, DEFAULT_MODEL_CONFIG.streaming);
  if (!rawStreaming || !isStreamingMode(rawStreaming)) {
    throw new Error(`Invalid streaming mode: ${rawStreaming ?? ''}`);
  }

  const model = firstDefined(cli.model, workspace.model?.model, process.env.CLIQ_MODEL, providerDef.getDefaultModel() ?? undefined);
  if (!model) {
    throw new Error(`model is required for provider ${provider}`);
  }

  const baseUrl = firstDefined(cli.baseUrl, workspace.model?.baseUrl, process.env.CLIQ_MODEL_BASE_URL, providerDef.defaultBaseUrl ?? undefined);
  if (!baseUrl) {
    throw new Error(`baseUrl is required for provider ${provider}`);
  }

  const apiKey = providerApiKey(provider);
  requireKey(provider, apiKey);

  return {
    provider,
    model,
    baseUrl,
    ...(apiKey ? { apiKey } : {}),
    streaming: rawStreaming
  };
}
```

function readModelConfig(record: Record<string, unknown>): PartialModelConfig | undefined {
  const value = record.model;
  if (value === undefined) {
    return undefined;
  }

  if (!value || typeof value !== 'object' || Array.isArray(value)) {
    throw new Error('model must be an object');
  }

  const model = value as Record<string, unknown>;
  for (const key of ['provider', 'model', 'baseUrl', 'streaming'] as const) {
    if (model[key] !== undefined && typeof model[key] !== 'string') {
      throw new Error(`model.${key} must be a string`);
    }
  }

  if (typeof model.streaming === 'string' && model.streaming !== 'auto' && model.streaming !== 'on' && model.streaming !== 'off') {
    throw new Error('model.streaming must be one of: auto, on, off');
  }

  return {
    ...(typeof model.provider === 'string' ? { provider: model.provider } : {}),
    ...(typeof model.model === 'string' ? { model: model.model } : {}),
    ...(typeof model.baseUrl === 'string' ? { baseUrl: model.baseUrl } : {}),
    ...(typeof model.streaming === 'string' ? { streaming: model.streaming } : {})
  };
}

import { MODEL_TIMEOUT_MS } from '../config.js';

export function joinUrl(baseUrl: string, path: string) {
  return `${baseUrl.replace(/\/+$/, '')}/${path.replace(/^\/+/, '')}`;
}

export async function fetchWithTimeout(url: string, init: RequestInit, timeoutMs = MODEL_TIMEOUT_MS) {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), timeoutMs);
  try {
    return await fetch(url, { ...init, signal: controller.signal });
  } catch (error) {
    if (error instanceof Error && error.name === 'AbortError') {
      throw new Error(`Model request timed out after ${timeoutMs}ms`);
    }
    throw error;
  } finally {
    clearTimeout(timeout);
  }
}

export async function readJsonResponse<T>(response: Response, providerName: string): Promise<T> {
  if (!response.ok) {
    throw new Error(`${providerName} error ${response.status}: ${await response.text()}`);
  }
  return (await response.json()) as T;
}

  }

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let output = '';

  for (;;) {
    const { done, value } = await reader.read();
    if (done) break;
    const text = decoder.decode(value, { stream: true });
    output += text;
    await onChunk(text);
  }

  output += decoder.decode();
  return output;
}

export async function readSseDeltas(response: Response, extractDelta: (json: unknown) => string | null, onDelta: (text: string) => void | Promise<void>) {
  let buffer = '';
  let content = '';

  await readTextStream(response, async (chunk) => {
    buffer += chunk;
    const frames = buffer.split('\n\n');
    buffer = frames.pop() ?? '';

    for (const frame of frames) {
      for (const line of frame.split('\n')) {
        if (!line.startsWith('data: ')) continue;
        const payload = line.slice('data: '.length).trim();
        if (!payload || payload === '[DONE]') continue;
        const delta = extractDelta(JSON.parse(payload));
        if (delta) {
          content += delta;
          await onDelta(delta);
        }
      }
    }
  });

  return content;
}

export async function readNdjsonDeltas(response: Response, extractDelta: (json: unknown) => string | null, onDelta: (text: string) => void | Promise<void>) {
  let buffer = '';
  let content = '';

  await readTextStream(response, async (chunk) => {
    buffer += chunk;
    const lines = buffer.split('\n');
    buffer = lines.pop() ?? '';

    for (const line of lines) {
      const payload = line.trim();
      if (!payload) continue;
      const delta = extractDelta(JSON.parse(payload));
      if (delta) {
        content += delta;
        await onDelta(delta);
      }
    }
  });

  return content;
}
